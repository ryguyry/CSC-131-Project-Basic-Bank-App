CREATE TABLE IOU (
    iou_id INT AUTO_INCREMENT PRIMARY KEY,
    lender_id INT,
    borrower_id INT,
    amount DECIMAL(10, 2) NOT NULL,
    status ENUM('pending', 'paid') DEFAULT 'pending',
    due_date DATE,
    FOREIGN KEY (lender_id) REFERENCES Users(user_id),
    FOREIGN KEY (borrower_id) REFERENCES Users(user_id)
);
